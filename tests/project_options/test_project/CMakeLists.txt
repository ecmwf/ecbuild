cmake_minimum_required( VERSION 3.18 FATAL_ERROR )
find_package( ecbuild 3.4 REQUIRED )

project( proja LANGUAGES NONE VERSION 0.2 )
ecbuild_add_option(
    FEATURE FEATUREON
    DEFAULT ON
    CONDITION DEFINED FEATUREON_CONDITION
)
ecbuild_add_option(
    FEATURE FEATUREOFF
    DEFAULT OFF
    CONDITION DEFINED FEATUREOFF_CONDITION
)
